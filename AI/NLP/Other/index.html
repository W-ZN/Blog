    DGL 库条件下进行知识抽取：

    import dgl

def extract_multi_hop(graph: dgl.DGLGraph, node_id: int, num_hops: int) -> dgl.DGLGraph:
    """
    从知识图谱中提取指定节点的多跳关系。

    Args:
        graph (dgl.DGLGraph): 存储知识图谱的图数据结构。
        node_id (int): 开始结点 ID。
        num_hops (int): 指定的跳数。

    Returns:
        dgl.DGLGraph: 存储提取出的子图的图数据结构。
    """
    # 定义一个子图对象，用于存储提取出的子图。
    subgraph = dgl.DGLGraph()

    # 定义一个队列，用于广度优先搜索。
    queue = [node_id]

    # 定义一个集合，用于存储已经被访问过的节点。
    visited = set([node_id])

    # 进行多跳搜索。
    for i in range(num_hops):
        # 保存当前队列的长度。
        cur_len = len(queue)

        # 逐个访问当前队列中的结点。
        for j in range(cur_len):
            # 从队列中取出一个结点。
            cur_node = queue.pop(0)

            # 将当前结点加入到子图中。
            subgraph.add_nodes(1)
            subgraph.nodes[i].data['id'] = torch.tensor([cur_node])

            # 遍历当前结点的邻居结点。
            neighbors = graph.successors(cur_node)
            for neighbor in neighbors:
                # 如果当前邻居结点还没有被访问过，则将其加入到队列和集合中。
                if neighbor not in visited:
                    visited.add(neighbor)
                    queue.append(neighbor)

                    # 在子图中添加邻居结点。
                    subgraph.add_nodes(1)
                    subgraph.nodes[i+1].data['id'] = torch.tensor([neighbor])
                    subgraph.add_edge(i, i+1)

    return subgraph

    import torch
import torch.nn.functional as F

# 定义三元组表示法，即(head, relation, tail)
class TripletsDataset(torch.utils.data.Dataset):
    def __init__(self, triplets):
        self.triplets = triplets

    def __getitem__(self, index):
        return self.triplets[index]

    def __len__(self):
        return len(self.triplets)

# 定义基于深度优先搜索的函数，用于提取指定结点的子图
def extract_subgraph(start_node, max_depth, triplets):
    # 初始化相关数据结构
    subgraph_triplets = []
    visited_nodes = set()
    queue = [(start_node, 0)]

    # 开始深度优先搜索
    while queue:
        node, depth = queue.pop(0)
        visited_nodes.add(node)

        # 判断是否达到指定深度，如果达到了就跳过
        if depth >= max_depth:
            continue

        # 遍历所有三元组
        for head, relation, tail in triplets:
            # 找到所有与当前结点相连的三元组
            if head == node:
                subgraph_triplets.append((head, relation, tail))

                # 将与当前结点相连的未访问结点加入队列中
                if tail not in visited_nodes:
                    queue.append((tail, depth + 1))
            elif tail == node:
                subgraph_triplets.append((head, relation, tail))

                # 将与当前结点相连的未访问结点加入队列中
                if head not in visited_nodes:
                    queue.append((head, depth + 1))

    # 返回子图的三元组列表
    return subgraph_triplets

# 定义模型类，用于表示知识图谱
class KnowledgeGraph(torch.nn.Module):
    def __init__(self, triplets):
        super(KnowledgeGraph, self).__init__()

           # 将三元组转化为PyTorch张量
    self.triplets = torch.tensor(triplets)

    # 提取所有结点的编号
    nodes = set(self.triplets[:, [0, 2]].flatten().tolist())
    self.node2index = {node: index for index, node in enumerate(nodes)}

    # 将三元组的头尾结点映射为编号
    self.triplets[:, [0, 2]] = self.triplets[:, [0, 2]].apply(lambda x: self.node2index[x])

# 定义前向传播函数
def forward(self, nodes, depth):
    # 提取指定结点的子图
    subgraph_triplets = []
    for node in nodes:
        subgraph_triplets += extract_subgraph(node, depth, self.triplets.tolist())

    # 将子图转化为PyTorch张量
    subgraph_triplets = torch.tensor(subgraph_triplets)

    # 将子图的头尾结点映射为编号
    subgraph_triplets[:, [0, 2]] = subgraph_triplets[:, [0, 2]].apply(lambda x: self.node2index[x])

    # 定义关系矩阵
    num_nodes = len(self.node2index)
    num_relations = len(set(self.triplets[:, 1]))
    relation_matrix = torch.zeros((num_relations, num_nodes, num_nodes))

    # 将子图的关系矩阵赋值给总的关系矩阵
    for head, relation, tail in subgraph_triplets:
        relation_matrix[relation, tail, head] = 1

    # 计算结点的嵌入向量
    node_embeddings = torch.zeros((num_nodes, 128))
    for i in range(num_relations):
        relation_embeddings = torch.randn(128, 128)
        node_embeddings = torch.matmul(relation_matrix[i], relation_embeddings)

    # 返回指定结点的嵌入向量
    node_embeddings = F.normalize(node_embeddings, p=2, dim=1)
    return node_embeddings[nodes, :]
    1. 定义了三元组表示法，即(head, relation, tail)，用于表示知识图谱中的关系。
    2. 定义了一个基于深度优先搜索的函数，用于提取指定结点的子图。
    3. 定义了一个模型类KnowledgeGraph，用于表示整个知识图谱。其中，模型的forward函数接受两个参数，即待提取子图的结点和指定的深度，返回指定结点的嵌入向量。
